💡 核心理解

  • 1️⃣ Loop 只有两个出口：LLM 回复纯文本（正常结束）或达到 max iterations（强制结束）
  • 2️⃣ Tool 执行是 loop 的唯一分支：有 tool_call → 执行 → 结果喂回 → 继续；没有 → 结束
  • 3️⃣ 并行优化：read-only 的 tool（如 FileRead、Glob）可以并行，write 的（如 Bash、FileEdit）必须串行
  • 4️⃣ 安全检查在执行前：不是所有 tool call 都会被执行，权限检查可以拦截
  • 5️⃣ Auto-compact 是透明的：用户感知不到，但确保 loop 不会因为 token 溢出而崩溃

// jesse-agent/src/loop.ts — 核心就这么多
async function agentLoop(userMessage: string, messages: Message[]) {
  messages.push({ role: "user", content: userMessage })

  while (true) {
    // Step 2: 调 LLM
    const response = await callLLM(messages, tools)

    // Step 4: 判断
    if (response.type === "text") {
      console.log(response.content)  // 结束
      break
    }

    // Step 5-7: 执行 tool，结果喂回
    for (const toolCall of response.toolCalls) {
      const result = await executeTool(toolCall.name, toolCall.args)
      messages.push({
        role: "tool",
        tool_call_id: toolCall.id,
        content: result
      })
    }

    // Step 8: 安全阀
    if (messages.length > MAX_ITERATIONS) break
  }
}
